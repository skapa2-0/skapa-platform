from openai import OpenAI

client = OpenAI(
    base_url="https://api.skapa.design/api/v1",
    api_key="VOTRE_CLE_SKAPA",
)

response = client.chat.completions.create(
    model="anthropic/claude-sonnet",  # Ou openai/gpt-4o, mistral/large, etc.
    messages=[
        {"role": "user", "content": "Bonjour ! Explique-moi l'IA en 3 phrases."}
    ],
)

print(response.choices[0].message.content)
